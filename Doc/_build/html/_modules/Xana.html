    def xpcs(self, series_id, saxs=None, filename="", handle_existing='next', **kwargs):
        for sid in series_id:
            print('\n\n#### Starting XPCS Analysis ####\nSeries: {} in folder {}\n'.format(sid, self.xdata.datdir))
            if type(sid) == int or type(sid) == np.int64:
                inputd = {'files':self.xdata.serieslist[sid],'format':self.xdata.fmtstr}
            elif type(sid) == tuple:
                sid, first, last = sid
                inputd = {'files':self.xdata.serieslist[sid], 'first':first, 'last':last,
                                 'format':self.xdata.fmtstr}
            else:
                pass
            
            Isaxs = saxs
            if saxs == 'get':
                Isaxs = self.saxs(sid, filename=filename, handle_existing=handle_existing,
                                      return_saxs=True)[0]['Isaxs']

            dt = self.xdata.meta.iloc[sid]['t_exposure'] +  self.xdata.meta.iloc[sid]['t_readout']
            corfd = pyxpcs(inputd, self.setup['qroi'], dt=dt, qv=self.setup['qv'], saxs=Isaxs,
                               mask = self.xdata.mask, ctr=self.setup['ctr'], prefix=self.xdata.prefix,
                               **kwargs)

            f = self.xdata.datdir.split('/')[-2] + '_s{:02d}'.format(sid) + filename
            savfile = save_result(corfd, 'xpcs', self.savdir, f, handle_existing=handle_existing)

            self.add_db_entry(sid, savfile)

    def saxs(self, series_id, filename="", handle_existing='next', return_saxs=True,
                 nprocs=8, verbose=True, **kwargs):
        for sid in series_id:
            print('\n#### Starting SAXS Analysis ####\nSeries: {} in folder {}\n'.format(sid, self.xdata.datdir))
            
            Isaxs, Vsaxs = self.xdata.get_series(sid, method='average', verbose=verbose, nprocs=nprocs)
            I, q = get_soq(Isaxs, self.xdata.mask, self.setup)
            soq = np.hstack((q,I))
            saxsd = {'Isaxs':Isaxs, 'Vsaxs':Vsaxs, 'soq':soq}

            f = self.xdata.datdir.split('/')[-2] + '_s{:02d}'.format(sid) + filename
            savfile = save_result(saxsd, 'saxs', self.savdir, f, handle_existing=handle_existing)
            self.add_db_entry(sid, savfile)
            if return_saxs:

                return saxsd, savfile

    def xsvs(self, series_id, saxs=None, filename="", handle_existing='next', nbins=15, verbose=True,
                 nprocs=4, **kwargs):
        
        for sid in series_id:
            print('\n\n#### Starting XSVS Analysis ####\nSeries: {} in folder {}\n'.format(sid, self.xdata.datdir))

            Isaxs = saxs
            if saxs == 'get':
                Isaxs = self.saxs(sid, filename=filename, handle_existing=handle_existing,
                                      return_saxs=True, verbose=verbose)[0]['Isaxs']
            xsvsd = {'Isaxs':Isaxs}

            t_exp = self.xdata.meta.iloc[sid]['t_exposure']
            xsvsd['t_exp'] = t_exp
            xsvsd['k'] = np.arange(nbins)
            xsvsd['qv'] = self.setup['qv']
            xsvsd['qroi'] = self.setup['qroi']
            xsvsd['xsvs'] = self.xdata.get_series(sid, verbose=verbose, method='probability', nprocs=nprocs,
                                              qroi=self.setup['qroi'], nbins=nbins)
            
            f = self.xdata.datdir.split('/')[-2] + '_s{:02d}'.format(sid) + filename
            savfile = save_result(xsvsd, 'xsvs', self.savdir, f, handle_existing=handle_existing)

            self.add_db_entry(sid, savfile)

[docs]    def load_db(self, dbfile=None, handle_existing='raise'):
        if dbfile is None:
            dbfile = self.dbfile
            print('Try loading default database {}'.format(dbfile))
        else:
            if '/' in dbfile:
                pass
            else:
                dbfile = self.savdir + dbfile
            print('Try loading database {}'.format(dbfile))
        try:
            self.db = pickle.load(open(dbfile, 'rb'))
            self.dbfile = dbfile
            print('Loaded analysis database from {}'.format(dbfile))
        except OSError:
            self.dbfile = self.savdir + os.path.split(dbfile)[-1]
            print('Analysis database not found.\nInitialize database...'.format(dbfile))

            self.init_db(dbfile, handle_existing)

    def discard_entry(self, db_id):
        discard = []
        for i in db_id:
            datdir = self.db.loc[i, 'datdir']
            series = self.db.loc[i, 'series']
            discard.append(self.db[(self.db['datdir'].str.match(datdir))&(self.db['series']==series)].index.values)
        for i in np.unique(np.hstack(discard)):
            self.db.loc[i,'use'] = False

        self.save_db(handle_existing='overwrite')

[docs]    def update_colors(self, cmap=None, factor=1, repeat=1,):
        factor = int(factor)
        if cmap is None:
            cmap = self.cmap
        else:
            self.cmap = cmap

        self.colors = []
        cm = plt.get_cmap(cmap)
        ci = np.linspace(0,1,factor)
        for i in range(factor):
            self.colors.append(cm(ci[i]))

        self.colors = np.tile(self.colors, (repeat, 1))
